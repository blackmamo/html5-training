  host: process.env.ELKSERVER_PORT_9200_TCP_ADDR + ":9200",
  log: "info"
});

  level: "info",
  transports: [
    new winston.transports.File({
      name: "loggerA",

    }),
    new winston.transports.File({
      name: "loggerB",

});
